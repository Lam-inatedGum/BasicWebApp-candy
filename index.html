What is your preference?

let score = 0; let user_prefs = [ prompt("Do you prefer sweeet or sour candy? "), prompt("Do you prefer fruity or chocolatey candy? "), prompt("Do you preffer hard or soft candy? ") ]; user_prefs.forEach(pref => { if (pref === "sweet" || pref === "fruity" || pref === "soft") { score++; } else if (pref === "sour" || pref === "chocolatey" || pref === "hard") { score--; } });
